-- Migration: 20260522000025_board_rls.sql
-- Phase 5 Board Management RLS + Storage policies (D-25, D-28, D-33 to D-35, D-32)
-- All policies use public.institution_id() helper for tables and JWT app_metadata for Storage.
-- Immutability: UPDATE/DELETE blocked when parent meeting.status = 'closed' via subquery.
-- Role checks mirror the RBAC matrix in CONTEXT D-28/D-29/D-30/D-31 exactly.
-- No policies for operations that must be denied (e.g. risk-officer cannot upload packs).

-- Enable RLS on all four tables
alter table public.board_meetings enable row level security;
alter table public.board_meeting_documents enable row level security;
alter table public.board_resolutions enable row level security;
alter table public.board_action_items enable row level security;

-- ============================================================
-- board_meetings policies (D-09, D-25, D-28)
-- ============================================================
create policy "board_meetings_select"
  on public.board_meetings for select
  using ( institution_id = (select public.institution_id()) );

create policy "board_meetings_insert"
  on public.board_meetings for insert
  with check (
    institution_id = (select public.institution_id())
    and (select public.active_role()) in ('admin', 'board-secretary')
  );

create policy "board_meetings_update"
  on public.board_meetings for update
  using (
    institution_id = (select public.institution_id())
    and status != 'closed'
    and (select public.active_role()) in ('admin', 'board-secretary')
  )
  with check (
    institution_id = (select public.institution_id())
    and status != 'closed'
    and (select public.active_role()) in ('admin', 'board-secretary')
  );

-- No DELETE policy (immutable records; use audit amendments only)

-- ============================================================
-- board_meeting_documents policies (D-25, D-15, D-28 view packs excludes risk-officer)
-- ============================================================
create policy "board_meeting_documents_select"
  on public.board_meeting_documents for select
  using (
    institution_id = (select public.institution_id())
    and (select public.active_role()) != 'risk-officer'
  );

create policy "board_meeting_documents_insert"
  on public.board_meeting_documents for insert
  with check (
    institution_id = (select public.institution_id())
    and (select public.active_role()) in ('admin', 'board-secretary')
    and (select status from public.board_meetings where id = meeting_id) != 'closed'
  );

create policy "board_meeting_documents_update"
  on public.board_meeting_documents for update
  using (
    institution_id = (select public.institution_id())
    and (select status from public.board_meetings where id = meeting_id) != 'closed'
    and (select public.active_role()) in ('admin', 'board-secretary')
  )
  with check ( same as using );

-- No DELETE (append-only reference material)

-- ============================================================
-- board_resolutions policies (D-19, D-25, D-30 RECORD_ROLES)
-- Only insert allowed while meeting in_progress
-- ============================================================
create policy "board_resolutions_select"
  on public.board_resolutions for select
  using ( institution_id = (select public.institution_id()) );

create policy "board_resolutions_insert"
  on public.board_resolutions for insert
  with check (
    institution_id = (select public.institution_id())
    and (select status from public.board_meetings where id = meeting_id) = 'in_progress'
    and (select public.active_role()) in ('admin', 'ceo', 'board-member', 'board-secretary')
  );

-- No UPDATE/DELETE (append-only per BOARD-06)

-- ============================================================
-- board_action_items policies (D-23, D-25, D-31 ACTION_UPDATE_ROLES + owner self-update)
-- ============================================================
create policy "board_action_items_select"
  on public.board_action_items for select
  using ( institution_id = (select public.institution_id()) );

create policy "board_action_items_insert"
  on public.board_action_items for insert
  with check (
    institution_id = (select public.institution_id())
    and (select status from public.board_meetings where id = meeting_id) != 'closed'

export function isMeetingClosed(status: MeetingStatus): boolean {

}

export function getEscalationThreshold(dueDateStr: string): 'none' | 'early_warning' | 'due_today' | 'critical_overdue' {
  const due = new Date(dueDateStr);
  const now = new Date();
  const diffDays = Math.floor((due.getTime() - now.getTime()) / (1000 * 3600 * 24));

  if (diffDays > 3) return 'none';
  if (diffDays >= 1 && diffDays <= 3) return 'early_warning';
  if (diffDays === 0) return 'due_today';
  if (diffDays <= -7) return 'critical_overdue';
  return 'early_warning'; // 1-6 days overdue
}

export function buildBoardPackStoragePath(
  institutionId: string,
  meetingId: string,
  originalFilename: string
): string {
  const timestamp = Math.floor(Date.now() / 1000);
  // sanitize per D-12 (spaces → _, strip special)
  const sanitized = originalFilename.replace(/[^a-zA-Z0-9.-]/g, '_').replace(/_{2,}/g, '_');
  return `${institutionId}/${meetingId}/${timestamp}_${sanitized}`;
}

export function computeBoardActionStats(actions: BoardActionItemRow[]) {
  const open = actions.filter(a => a.status !== 'completed' && a.status !== 'cancelled').length;
  const overdue = actions.filter(a => {
    const due = new Date(a.due_date);
    return due < new Date() && a.status !== 'completed' && a.status !== 'cancelled';
  }).length;
  return { open, overdue, total: actions.length };
}
