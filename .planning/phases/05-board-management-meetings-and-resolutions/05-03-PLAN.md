```ts
import { type NextRequest } from 'next/server'
import { createClient } from '@/lib/supabase/server'
import { verifyChecksum } from '@/lib/files/checksum'
import type { AppRole } from '@/types/auth'

const VIEW_ROLES: AppRole[] = ['admin', 'ceo', 'compliance-officer', 'risk-officer', 'audit-officer', 'board-member']

export async function GET(_request: NextRequest, { params }: { params: { id: string } }) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) return new Response('Unauthorized', { status: 401 })

  const activeRole = (user.app_metadata as any)?.active_role
  if (!activeRole || !VIEW_ROLES.includes(activeRole as AppRole)) return new Response('Forbidden', { status: 403 })

  const { data: row } = await supabase.from('obligation_evidence').select('storage_path, sha256_hash, original_filename, mime_type').eq('id', params.id).single()
  if (!row) return new Response('Not Found', { status: 404 })

  const { data: fileData, error: dlError } = await supabase.storage.from('compliance-evidence').download(row.storage_path)
  if (dlError || !fileData) return new Response('Storage error', { status: 500 })

  const buffer = Buffer.from(await fileData.arrayBuffer())
  const verified = await verifyChecksum(buffer, row.sha256_hash)
  if (!verified) return new Response('Integrity check failed (SHA-256 mismatch)', { status: 409 })

  const filename = encodeURIComponent(row.original_filename)
  return new Response(buffer, {
    headers: {
      'Content-Type': row.mime_type,
      'Content-Disposition': `attachment; filename*=UTF-8''${filename}`,
      'Content-Length': String(buffer.length),
    },
  })
}
```

```ts
// app/api/board/documents/[id]/route.ts
// GET — download board pack from private 'board-packs' Storage with SHA-256 integrity verification (D-16, TRAIL-04)
// Even though board packs are not legal evidence, integrity check prevents silent corruption.
// Role matrix: excludes risk-officer per D-28 (view packs ❌ for risk)
import { type NextRequest } from 'next/server'
import { createClient } from '@/lib/supabase/server'
import { verifyChecksum } from '@/lib/files/checksum'
import type { AppRole } from '@/types/auth'

// VIEW roles that may download board packs (matrix D-28: risk-officer excluded)
const VIEW_ROLES: AppRole[] = [
  'admin',
  'ceo',
  'board-member',
  'board-secretary',
  'audit-officer',
]

export async function GET(
  _request: NextRequest,
  { params }: { params: { id: string } }
) {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) {
    return new Response('Unauthorized', { status: 401 })
  }

  const appMeta = user.app_metadata as Record<string, string>
  const activeRole = appMeta?.active_role as AppRole | undefined
  if (!activeRole || !VIEW_ROLES.includes(activeRole)) {
    return new Response('Forbidden', { status: 403 })
  }

  // Fetch document metadata (RLS already scoped to institution)
  const { data: doc, error: dbErr } = await supabase
    .from('board_meeting_documents')
    .select('storage_path, sha256_hash, original_filename, mime_type, file_size_bytes')
    .eq('id', params.id)
    .single()

  if (dbErr || !doc) {
    return new Response('Not Found', { status: 404 })
  }

  // Download from private bucket
  const { data: fileData, error: storageErr } = await supabase.storage
    .from('board-packs')
    .download(doc.storage_path)

  if (storageErr || !fileData) {
    return new Response('Storage download failed', { status: 500 })
  }

  const buffer = Buffer.from(await fileData.arrayBuffer())

  // TRAIL-04 integrity check (even for non-evidence packs)
  const verified = await verifyChecksum(buffer, doc.sha256_hash)
  if (!verified) {
    return new Response('Integrity check failed — file may have been modified (SHA-256 mismatch)', { status: 409 })
  }

  // RFC 5987 filename encoding (CR-01 pre-empted, same as Phase 4)
  const filename = encodeURIComponent(doc.original_filename)
  const contentDisposition = `attachment; filename*=UTF-8''${filename}`

  return new Response(buffer, {
    status: 200,
    headers: {
      'Content-Type': doc.mime_type,
      'Content-Disposition': contentDisposition,
      'Content-Length': String(buffer.length),
      'Cache-Control': 'private, max-age=3600',
    },
  })
}
```
