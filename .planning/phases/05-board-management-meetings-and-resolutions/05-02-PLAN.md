// lib/board/queries.ts
// Query helpers for board meetings, documents, resolutions, action items, dashboard stats, and cron escalation (D-44, D-46).
import type { SupabaseClient } from '@supabase/supabase-js'
import type { Database } from '@/types/supabase'
import type {
  BoardMeetingRow,
  BoardResolutionRow,
  BoardActionItemRow,
  BoardDocumentRow,
  MeetingStatus,
  ActionItemStatus,
} from '@/types/board'

type DbClient = SupabaseClient<Database>

export interface ActionItemEscalationTarget {
  id: string
  title: string
  due_date: string
  owner_id: string
  meeting_id: string
  meeting_title: string
  institution_id: string
}

export async function listMeetings(
  supabase: DbClient,
  options?: { limit?: number; upcoming?: boolean; status?: MeetingStatus }
): Promise<BoardMeetingRow[]> {
  let query = supabase
    .from('board_meetings')
    .select(`
      id, title, meeting_date, location, status, agenda_items, attendee_ids,
      created_by, closed_at, created_at, updated_at
    `)
    .order('meeting_date', { ascending: true })

  if (options?.upcoming) {
    query = query.gte('meeting_date', new Date().toISOString()).neq('status', 'closed')
  }
  if (options?.status) {
    query = query.eq('status', options.status)
  }
  if (options?.limit) query = query.limit(options.limit)

  const { data, error } = await query
  if (error) throw error
  return (data ?? []) as BoardMeetingRow[]
}

export async function getMeetingById(supabase: DbClient, id: string): Promise<BoardMeetingRow | null> {
  const { data, error } = await supabase
    .from('board_meetings')
    .select(`*`)
    .eq('id', id)
    .single()
  if (error) return null
  return data as BoardMeetingRow
}

export async function listDocuments(supabase: DbClient, meetingId: string): Promise<BoardDocumentRow[]> {
  const { data } = await supabase
    .from('board_meeting_documents')
    .select(`*, user_profiles!uploaded_by(first_name, last_name)`)
    .eq('meeting_id', meetingId)
    .order('uploaded_at', { ascending: false })
  return (data ?? []).map((d: any) => ({
    ...d,
    uploaded_by_name: d.user_profiles ? `${d.user_profiles.first_name} ${d.user_profiles.last_name}` : null,
  })) as BoardDocumentRow[]
}

export async function listResolutions(supabase: DbClient, meetingId: string): Promise<BoardResolutionRow[]> {
  const { data } = await supabase
    .from('board_resolutions')
    .select(`
      *,
      proposer:user_profiles!proposer_id(first_name, last_name),
      seconder:user_profiles!seconder_id(first_name, last_name)
    `)
    .eq('meeting_id', meetingId)
    .order('resolution_number', { ascending: true })
  return (data ?? []).map((r: any) => ({
    ...r,
    proposer_name: r.proposer ? `${r.proposer.first_name} ${r.proposer.last_name}` : '',
    seconder_name: r.seconder ? `${r.seconder.first_name} ${r.seconder.last_name}` : null,
  })) as BoardResolutionRow[]
}

export async function listActionItems(supabase: DbClient, meetingId?: string): Promise<BoardActionItemRow[]> {
  let q = supabase
    .from('board_action_items')
    .select(`
      *,
      owner:user_profiles!owner_id(first_name, last_name),
      meeting:board_meetings!meeting_id(title)
    `)
    .order('due_date', { ascending: true })
  if (meetingId) q = q.eq('meeting_id', meetingId)
  const { data } = await q
  return (data ?? []).map((a: any) => ({
    ...a,
    owner_name: a.owner ? `${a.owner.first_name} ${a.owner.last_name}` : '',
    meeting_title: a.meeting?.title,
  })) as BoardActionItemRow[]
}

export async function getBoardStats(supabase: DbClient) {
  const now = new Date()
  const thirtyDays = new Date(now.getTime() + 30 * 86400000).toISOString()

  const [upcoming, openActions, overdue] = await Promise.all([
    supabase
      .from('board_meetings')
      .select('id', { count: 'exact', head: true })
      .gte('meeting_date', now.toISOString())
      .neq('status', 'closed'),
    supabase
      .from('board_action_items')
      .select('id', { count: 'exact', head: true })
      .neq('status', 'completed')
      .neq('status', 'cancelled'),
    supabase
      .from('board_action_items')
      .select('id', { count: 'exact', head: true })
      .lt('due_date', now.toISOString().split('T')[0])
      .neq('status', 'completed')
      .neq('status', 'cancelled'),
  ])

  return {
    upcomingMeetings: upcoming.count ?? 0,
    openActionItems: openActions.count ?? 0,
    overdueActions: overdue.count ?? 0,
  }
}

export async function getActionItemsForEscalation(supabase: DbClient): Promise<ActionItemEscalationTarget[]> {
  const { data } = await supabase
    .from('board_action_items')
    .select(`
      id, title, due_date, owner_id, meeting_id,
      meeting:board_meetings!inner(title, institution_id)
    `)
    .lt('due_date', new Date().toISOString().split('T')[0])
    .neq('status', 'completed')
    .neq('status', 'cancelled')

  return (data ?? []).map((row: any) => ({
    id: row.id,
    title: row.title,
    due_date: row.due_date,
    owner_id: row.owner_id,
    meeting_id: row.meeting_id,
    meeting_title: row.meeting.title,
    institution_id: row.meeting.institution_id,
  }))
}

// lib/board/escalation.ts
// Board action item escalation emails (D-36–D-39). Standalone module, escapeHtml from Day 1 (ME-04).
import { Resend } from 'resend'
import { createAdminClient } from '@/lib/supabase/admin'
import { getEscalationThreshold } from '@/lib/board/board-utils'
import type { ActionItemEscalationTarget } from '@/lib/board/queries'

function escapeHtml(str: string): string {
  return str
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
}

const resend = new Resend(process.env.RESEND_API_KEY!)

export async function sendBoardActionEscalationEmails(targets: ActionItemEscalationTarget[]) {
  if (!targets.length) return
  const admin = createAdminClient()
  const thresholds = targets.map(t => getEscalationThreshold(t.due_date))

  for (let i = 0; i < targets.length; i++) {
    const t = targets[i]
    const thresh = thresholds[i]
    if (thresh === 'none') continue

    // Lookup owner + board-secretary + admin emails via admin client (bypasses RLS for cron)
    const { data: owner } = await admin.auth.admin.getUserById(t.owner_id)
    const ownerEmail = owner?.user?.email

    const { data: secretaries } = await admin
      .from('user_roles')
      .select('user_id')
      .in('role_name', ['board-secretary', 'admin'])
      .eq('institution_id', t.institution_id)  // assumes user_roles has institution_id

    const emails: string[] = []
    if (ownerEmail) emails.push(ownerEmail)
    // fetch emails for secretaries via getUserById in loop (simplified)
    for (const s of secretaries ?? []) {
      const { data: u } = await admin.auth.admin.getUserById(s.user_id)
      if (u?.user?.email) emails.push(u.user.email)
    }
    const uniqueEmails = [...new Set(emails)]

    const subject = thresh === 'critical_overdue'
      ? `CRITICAL: Overdue Board Action Item — ${escapeHtml(t.title)}`
      : `Board Action Reminder: ${escapeHtml(t.title)}`

    const html = `
      <div style="font-family: DM Sans, sans-serif; max-width: 600px;">
        <h2 style="color: #C8A44A;">GRC-Nexus Board Escalation</h2>
        <p>Action item <strong>${escapeHtml(t.title)}</strong> from meeting <strong>${escapeHtml(t.meeting_title)}</strong> is ${thresh.replace('_', ' ')}.</p>
        <p><strong>Due:</strong> ${t.due_date}</p>
        <p>Please log in to GRC-Nexus to update status or complete the item.</p>
        <p style="color:#666;font-size:12px;">This notification was sent to the action owner and board secretaries/admins per institutional policy.</p>
      </div>
    `

    await resend.emails.send({
      from: 'GRC-Nexus Alerts <alerts@grcnexus.app>',
      to: uniqueEmails,
      subject,
      html,
    })
  }
}
