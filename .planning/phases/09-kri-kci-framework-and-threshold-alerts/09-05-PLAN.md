**Step 1: Add getKciDashboardStats to lib/audit/kci-queries.ts if not present**

If `getKciDashboardStats` was not written in 09-02, add it now to `lib/audit/kci-queries.ts`:

```typescript
export async function getKciDashboardStats(supabase: SupabaseClient) {
  const { data, error } = await supabase
    .from('kci_readings')
    .select('status')
    .order('recorded_at', { ascending: false })

  if (error) return { on_track: 0, at_risk: 0, breached: 0, no_data: 0 }

  const counts = { on_track: 0, at_risk: 0, breached: 0, no_data: 0 }
  for (const r of data ?? []) {
    const s = (r as { status: string }).status
    if (s in counts) counts[s as keyof typeof counts]++
  }
  return counts
}
```
