// lib/risk/kri-utils.ts
import type { IndicatorStatus, IndicatorDirection } from '@/types/kri'
export type { IndicatorStatus, IndicatorDirection }

/**
 * Calculate indicator status from actual vs target vs alert_threshold.
 *
 * lower_is_worse (e.g. pass rate %):
 *   actual >= target          → on_track
 *   actual >= alert_threshold → at_risk
 *   actual <  alert_threshold → breached
 *
 * higher_is_worse (e.g. incident count):
 *   actual <= target          → on_track
 *   actual <= alert_threshold → at_risk
 *   actual >  alert_threshold → breached
 */
export function calculateIndicatorStatus(
  actualValue: number | null | undefined,
  targetValue: number,
  alertThreshold: number,
  direction: IndicatorDirection,
): IndicatorStatus {
  if (actualValue === null || actualValue === undefined) return 'no_data'

  if (direction === 'lower_is_worse') {
    if (actualValue >= targetValue)    return 'on_track'
    if (actualValue >= alertThreshold) return 'at_risk'
    return 'breached'
  } else {
    if (actualValue <= targetValue)    return 'on_track'
    if (actualValue <= alertThreshold) return 'at_risk'
    return 'breached'
  }
}

/** Returns true when actual_value crosses alert_threshold in the configured direction. */
export function isIndicatorBreach(
  actualValue: number,
  alertThreshold: number,
  direction: IndicatorDirection,
): boolean {
  return direction === 'lower_is_worse'
    ? actualValue < alertThreshold
    : actualValue > alertThreshold
}

export const INDICATOR_STATUS_BADGE: Record<IndicatorStatus, { label: string; className: string }> = {
  on_track: { label: 'On Track',  className: 'bg-ok/10 text-ok border-ok/30' },
  at_risk:  { label: 'At Risk',   className: 'bg-warn/10 text-warn border-warn/30' },
  breached: { label: 'Breached',  className: 'bg-err/10 text-err border-err/30' },
  no_data:  { label: 'No Data',   className: 'bg-paper text-navy-mid border-paper-border' },

**lib/risk/kri-actions.ts** — Server actions following lib/strategic/actions.ts pattern exactly. Key differences: table is `kri_definitions`/`kri_readings`; roles are KRI-specific; recordKriReading MUST compute and persist status:

```typescript
'use server'
// lib/risk/kri-actions.ts
// SECURITY: Zod safeParse before any DB call — strips unknown fields.
// SECURITY: Role checked via user.app_metadata (JWT claim) — not DB query.
// SECURITY: institutionId sourced ONLY from user.app_metadata.institution_id.

import { revalidatePath } from 'next/cache'
import { createClient } from '@/lib/supabase/server'
import { kriDefinitionSchema, kriReadingSchema, type KriDefinitionInput, type KriReadingInput } from '@/lib/schemas/kri'
import { calculateIndicatorStatus } from '@/lib/risk/kri-utils'
import type { AppRole } from '@/types/auth'

const KRI_WRITE_ROLES: AppRole[] = ['admin', 'ceo', 'risk-officer']
const GENERIC_ERROR = 'An unexpected error occurred. If this persists, contact your administrator.'

export async function createKriDefinition(
  values: KriDefinitionInput,
): Promise<{ error: string } | { data: { id: string } }> {
  const parsed = kriDefinitionSchema.safeParse(values)
  if (!parsed.success) {
    return { error: parsed.error.issues[0]?.message ?? 'Invalid input.' }
  }

  try {
    const supabase = await createClient()
    const { data: { user } } = await supabase.auth.getUser()
    if (!user) return { error: 'Unauthorized.' }

    const appMeta       = user.app_metadata as Record<string, string>
    const activeRole    = appMeta?.active_role as AppRole | undefined
    const institutionId = appMeta?.institution_id

    if (!activeRole || !KRI_WRITE_ROLES.includes(activeRole)) {
      return { error: 'You do not have permission to create KRI definitions.' }
    }

    const { data, error } = await supabase
      .from('kri_definitions')
      .insert({ ...parsed.data, institution_id: institutionId, created_by: user.id })
      .select('id')
      .single()

    if (error) {
      console.error('[createKriDefinition] DB error:', error)
      return { error: GENERIC_ERROR }
    }

    revalidatePath('/risk/kris')
    revalidatePath('/risk')
    revalidatePath('/dashboard')

    return { data: { id: (data as { id: string }).id } }
  } catch (err) {
    console.error('[createKriDefinition] Unexpected error:', err)
    return { error: GENERIC_ERROR }
  }
}

export async function recordKriReading(
  kriId: string,
  values: KriReadingInput,
): Promise<{ error: string } | { data: { id: string } }> {
  const parsed = kriReadingSchema.safeParse(values)
  if (!parsed.success) {
    return { error: parsed.error.issues[0]?.message ?? 'Invalid input.' }
  }

  try {
    const supabase = await createClient()
    const { data: { user } } = await supabase.auth.getUser()
    if (!user) return { error: 'Unauthorized.' }

    const appMeta    = user.app_metadata as Record<string, string>
    const activeRole = appMeta?.active_role as AppRole | undefined

    const { data: kri, error: kriError } = await supabase
      .from('kri_definitions')
      .select('owner_id, institution_id, target_value, alert_threshold, direction')
      .eq('id', kriId)
      .single()

    if (kriError || !kri) return { error: 'KRI definition not found.' }

    const kriRow = kri as { owner_id: string; institution_id: string; target_value: number; alert_threshold: number; direction: string }
    if (kriRow.owner_id !== user.id && activeRole !== 'admin') {
      return { error: 'Only the KRI owner or an administrator can record readings.' }
    }

    // CRITICAL: Compute and persist status — never rely on DB default 'no_data'.
    const computedStatus = calculateIndicatorStatus(
      parsed.data.actual_value,
      kriRow.target_value,
      kriRow.alert_threshold,
      kriRow.direction as 'lower_is_worse' | 'higher_is_worse',
    )

    const { data, error } = await supabase
      .from('kri_readings')
      .insert({
        ...parsed.data,
        kri_id: kriId,
        institution_id: kriRow.institution_id,
        recorded_by: user.id,
        status: computedStatus,
      })
      .select('id')
      .single()

    if (error) {
      console.error('[recordKriReading] DB error:', error)
      return { error: GENERIC_ERROR }
    }

    revalidatePath(`/risk/kris/${kriId}`)
    revalidatePath('/risk/kris')
    revalidatePath('/dashboard')

    return { data: { id: (data as { id: string }).id } }
  } catch (err) {
    console.error('[recordKriReading] Unexpected error:', err)
    return { error: GENERIC_ERROR }
  }
}
```

// lib/risk/kri-queries.ts
import type { SupabaseClient } from '@supabase/supabase-js'
import type { KriDefinitionWithReadings, KriReading } from '@/types/kri'

export const KRI_PAGE_SIZE = 20

export async function getKrisWithReadings(
  supabase: SupabaseClient,
  { page = 1 }: { page?: number } = {},
) {
  const from = (page - 1) * KRI_PAGE_SIZE
  const to   = from + KRI_PAGE_SIZE - 1

  const { data, count, error } = await supabase
    .from('kri_definitions')
    .select(
      `id, title, description, unit_of_measure, target_value, alert_threshold,
       direction, owner_id, reporting_frequency, risk_id, created_at, updated_at,
       kri_readings ( id, actual_value, period_start, period_end, status, recorded_at ),
       risks ( id, title ),
       user_profiles!owner_id ( first_name, last_name )`,
      { count: 'exact' },
    )
    .order('created_at', { ascending: false })
    .range(from, to)

       user_profiles!owner_id ( first_name, last_name )`,

export function getLatestKriReading(
  readings: KriReading[],
): KriReading | null {
  if (!readings || readings.length === 0) return null
  return [...readings].sort(
    (a, b) => new Date(b.recorded_at).getTime() - new Date(a.recorded_at).getTime(),
  )[0]
}

export async function getKriDashboardStats(supabase: SupabaseClient) {
  const { data, error } = await supabase
    .from('kri_readings')
    .select('status')
    .order('recorded_at', { ascending: false })

  if (error) return { on_track: 0, at_risk: 0, breached: 0, no_data: 0 }

  // Latest reading per KRI: PostgREST returns all readings; aggregate in app
  // For dashboard stat tiles, count readings by status
  const counts = { on_track: 0, at_risk: 0, breached: 0, no_data: 0 }
  for (const r of data ?? []) {
    const s = (r as { status: string }).status
    if (s in counts) counts[s as keyof typeof counts]++
  }
  return counts
}

// lib/risk/kri-alert.ts
// SECURITY: Uses createAdminClient() — RLS client cannot query auth.users.
// SECURITY: escapeHtml() applied to all user-supplied strings in email HTML.
// DEDUP: Only queries readings with recorded_at > (now() - interval '25 hours').

import { Resend } from 'resend'
import { createAdminClient } from '@/lib/supabase/admin'
import { isIndicatorBreach } from '@/lib/risk/kri-utils'

function escapeHtml(str: string): string {
  return str
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
}

const resend = new Resend(process.env.RESEND_API_KEY)

export async function sendKriBreachAlerts(): Promise<{ sent: number; skipped: number }> {
  const admin = createAdminClient()
  let sent = 0; let skipped = 0

  // Only look at readings recorded in the last 25 hours — prevents re-alerting on unchanged breaches.
  const cutoff = new Date(Date.now() - 25 * 60 * 60 * 1000).toISOString()

  const { data: recentReadings, error: readingsError } = await admin
    .from('kri_readings')
    .select('kri_id, actual_value, recorded_at')
    .gte('recorded_at', cutoff)

  if (readingsError) {
    console.error('[sendKriBreachAlerts] Failed to query recent readings:', readingsError)
    return { sent: 0, skipped: 0 }
  }

  // Group latest reading per KRI
  const latestByKri = new Map<string, { actual_value: number; recorded_at: string }>()
  for (const r of recentReadings ?? []) {
    const row = r as { kri_id: string; actual_value: number; recorded_at: string }
    const existing = latestByKri.get(row.kri_id)
    if (!existing || new Date(row.recorded_at) > new Date(existing.recorded_at)) {
      latestByKri.set(row.kri_id, { actual_value: row.actual_value, recorded_at: row.recorded_at })
    }
  }

  if (latestByKri.size === 0) return { sent: 0, skipped: 0 }

  // Fetch definitions for the KRIs that had recent readings
  const kriIds = Array.from(latestByKri.keys())
  const { data: definitions } = await admin
    .from('kri_definitions')
    .select('id, title, unit_of_measure, alert_threshold, direction, owner_id, institution_id')
    .in('id', kriIds)

  for (const def of definitions ?? []) {
    const d = def as { id: string; title: string; unit_of_measure: string; alert_threshold: number; direction: string; owner_id: string | null; institution_id: string }
    const reading = latestByKri.get(d.id)
    if (!reading) { skipped++; continue }

    if (!isIndicatorBreach(reading.actual_value, d.alert_threshold, d.direction as 'lower_is_worse' | 'higher_is_worse')) {
      skipped++; continue
    }

    const recipients = new Set<string>()

    if (d.owner_id) {
      const { data: ownerUser } = await admin.auth.admin.getUserById(d.owner_id)
      if (ownerUser?.user?.email) recipients.add(ownerUser.user.email)
    }

    const { data: govProfiles } = await admin
      .from('user_profiles')
      .select('id')
      .eq('institution_id', d.institution_id)
      .in('active_role', ['admin', 'ceo', 'risk-officer'])

    for (const profile of govProfiles ?? []) {
      const { data: u } = await admin.auth.admin.getUserById((profile as { id: string }).id)
      if (u?.user?.email) recipients.add(u.user.email)
    }

    if (recipients.size === 0) { skipped++; continue }

    try {
      await resend.emails.send({
        from: process.env.RESEND_FROM_EMAIL ?? 'noreply@grc-nexus.gov.zw',
        to: Array.from(recipients),
        subject: `⚠ KRI Alert: ${escapeHtml(d.title)} has breached threshold (${reading.actual_value} vs threshold ${d.alert_threshold})`,
        html: `
          <p>This is an automated KRI threshold alert from GRC-Nexus.</p>
          <p><strong>Indicator:</strong> ${escapeHtml(d.title)}</p>
          <p><strong>Actual Value:</strong> ${reading.actual_value} ${escapeHtml(d.unit_of_measure)}</p>
          <p><strong>Alert Threshold:</strong> ${d.alert_threshold} ${escapeHtml(d.unit_of_measure)}</p>
          <p>Please log in to GRC-Nexus to review this indicator and take corrective action.</p>
        `,
      })
      sent++
    } catch (emailErr) {
      console.error('[sendKriBreachAlerts] Email send error:', emailErr)
      skipped++
    }
  }

  return { sent, skipped }
}
