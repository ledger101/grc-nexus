```sql
-- Migration: 20260525000001_kri_schema.sql
-- Phase 9 KRI Framework: shared indicator enums + kri_definitions + kri_readings.
-- SECURITY: RLS enabled in 20260525000002_kri_rls.sql.
-- NOTE: Reuses public.kpi_frequency enum from Phase 2 strategic schema.

-- ============================================================
-- Shared Indicator Enums (used by both KRI and KCI)
-- ============================================================

create type public.indicator_status as enum (
  'on_track',
  'at_risk',
  'breached',
  'no_data'
);

create type public.indicator_direction as enum (
  'lower_is_worse',   -- e.g. pass rate: lower = worse; breach when actual < threshold
  'higher_is_worse'   -- e.g. incident count: higher = worse; breach when actual > threshold
);

-- ============================================================
-- kri_definitions
-- ============================================================

create table public.kri_definitions (
  id                  uuid primary key default gen_random_uuid(),
  institution_id      uuid not null references public.institutions(id) on delete restrict,
  risk_id             uuid references public.risks(id) on delete set null,
  title               text not null,
  description         text,
  unit_of_measure     text not null,
  target_value        numeric not null,
  alert_threshold     numeric not null,
  direction           public.indicator_direction not null default 'lower_is_worse',
  owner_id            uuid references public.user_profiles(id) on delete set null,
  reporting_frequency public.kpi_frequency not null default 'monthly',
  created_by          uuid references public.user_profiles(id) on delete set null,
  created_at          timestamptz not null default now(),
  updated_at          timestamptz not null default now()
);

create index idx_kri_definitions_institution_id on public.kri_definitions (institution_id);
create index idx_kri_definitions_risk_id        on public.kri_definitions (risk_id);
create index idx_kri_definitions_owner_id       on public.kri_definitions (owner_id);

-- ============================================================
-- kri_readings (append-only — no UPDATE RLS policy)
-- ============================================================

create table public.kri_readings (
  id             uuid primary key default gen_random_uuid(),
  institution_id uuid not null references public.institutions(id) on delete restrict,
  kri_id         uuid not null references public.kri_definitions(id) on delete cascade,
  period_start   date not null,
  period_end     date not null,
  actual_value   numeric not null,
  status         public.indicator_status not null default 'no_data',
  notes          text,
  recorded_by    uuid references public.user_profiles(id) on delete set null,
  recorded_at    timestamptz not null default now()
);

create index idx_kri_readings_institution_id on public.kri_readings (institution_id);
create index idx_kri_readings_kri_id         on public.kri_readings (kri_id);
create index idx_kri_readings_recorded_at    on public.kri_readings (recorded_at desc);
```

**20260525000002_kri_rls.sql** — Write this file with exact content:

```sql
-- Migration: 20260525000002_kri_rls.sql
-- RLS for kri_definitions (role-gated write) and kri_readings (owner/admin write; append-only).

-- ============================================================
-- kri_definitions
-- ============================================================

alter table public.kri_definitions enable row level security;
alter table public.kri_definitions force row level security;

create policy "kri_definitions_select" on public.kri_definitions
  for select to authenticated
  using (institution_id = (select public.institution_id()));

create policy "kri_definitions_insert" on public.kri_definitions
  for insert to authenticated
  with check (
    institution_id = (select public.institution_id())
    and (select public.active_role()) in ('admin', 'ceo', 'risk-officer')
  );

create policy "kri_definitions_update" on public.kri_definitions
  for update to authenticated
  using (institution_id = (select public.institution_id()))
  with check (
    institution_id = (select public.institution_id())
    and (select public.active_role()) in ('admin', 'ceo', 'risk-officer')
  );

-- ============================================================
-- kri_readings (no UPDATE policy — append-only for immutability)
-- ============================================================

alter table public.kri_readings enable row level security;
alter table public.kri_readings force row level security;

create policy "kri_readings_select" on public.kri_readings
  for select to authenticated
  using (institution_id = (select public.institution_id()));

create policy "kri_readings_insert" on public.kri_readings
  for insert to authenticated
  with check (
    institution_id = (select public.institution_id())
    and (
      (select public.active_role()) = 'admin'
      or recorded_by = (select public.user_profiles.id from public.user_profiles where public.user_profiles.id = auth.uid() limit 1)
    )
  );
```
