```typescript
import Link from 'next/link'
import { redirect } from 'next/navigation'
import { createClient } from '@/lib/supabase/server'
import { getKcisWithReadings, getLatestKciReading } from '@/lib/audit/kci-queries'
import { calculateIndicatorStatus } from '@/lib/audit/kci-utils'
import { IndicatorStatusBadge } from '@/components/indicators/IndicatorStatusBadge'
import { IndicatorSparkline } from '@/components/indicators/IndicatorSparkline'
import type { AppRole } from '@/types/auth'

export const dynamic = 'force-dynamic'
export const metadata = { title: 'Key Control Indicators — GRC-Nexus' }

const VIEW_ROLES: AppRole[] = ['admin', 'ceo', 'risk-officer', 'audit-officer', 'board-member', 'dept-head', 'compliance-officer', 'board-secretary']
const CREATE_ROLES: AppRole[] = ['admin', 'audit-officer']

export default async function KciListPage() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) redirect('/login')

  const appMeta    = user.app_metadata as Record<string, string>

  return (
    <div>
      <div className="mb-6 flex flex-wrap items-center justify-between gap-4">
        <div>
          <h1 className="text-[20px] font-semibold text-navy-900">Key Control Indicators</h1>
          <p className="mt-1 text-[14px] text-navy-mid">{kcis.length} indicator{kcis.length !== 1 ? 's' : ''} defined</p>
        </div>
        {canCreate && (
          <Link href="/audit/kcis/new" className="inline-flex items-center rounded-[8px] bg-gold px-4 py-2 text-[13px] font-medium text-navy-950 hover:bg-gold-hi">
            Add KCI
          </Link>
        )}
      </div>

      {kcis.length === 0 ? (
        <div className="rounded-[10px] border border-paper-border bg-white p-12 text-center shadow-card">
          <p className="text-[14px] text-navy-mid">No Key Control Indicators defined yet.</p>
          {canCreate && (
            <Link href="/audit/kcis/new" className="mt-4 inline-flex items-center rounded-[8px] bg-gold px-4 py-2 text-[13px] font-medium text-navy-950 hover:bg-gold-hi">
              Add the first KCI
            </Link>
          )}
        </div>
      ) : (
        <div className="overflow-hidden rounded-[10px] border border-paper-border bg-white shadow-card">
          <table className="w-full">
            <thead>
              <tr className="bg-paper border-b border-paper-border">
                <th className="text-left text-[13px] font-semibold uppercase tracking-wider text-navy-mid px-4 py-3">Name</th>
                <th className="text-left text-[13px] font-semibold uppercase tracking-wider text-navy-mid px-4 py-3">Unit</th>
                <th className="text-left text-[13px] font-semibold uppercase tracking-wider text-navy-mid px-4 py-3">Target / Threshold</th>
                <th className="text-left text-[13px] font-semibold uppercase tracking-wider text-navy-mid px-4 py-3">Status</th>
                <th className="text-left text-[13px] font-semibold uppercase tracking-wider text-navy-mid px-4 py-3">Trend</th>
              </tr>
            </thead>
            <tbody>
              {kcis.map((kci) => {
                const latest = getLatestKciReading(kci.kci_readings ?? [])

                  <tr key={kci.id} className="border-b border-paper-border last:border-0 hover:bg-paper/50">

                      <Link href={`/audit/kcis/${kci.id}`} className="text-[13px] font-medium text-navy-900 hover:text-gold">
                        {kci.title}
                      </Link>
                    </td>
                    <td className="px-4 py-3 text-[13px] text-navy-mid">{kci.unit_of_measure}</td>
                    <td className="px-4 py-3 text-[13px] font-mono text-navy-900">{kci.target_value} / {kci.alert_threshold}</td>
                    <td className="px-4 py-3"><IndicatorStatusBadge status={status} /></td>
                    <td className="px-4 py-3"><IndicatorSparkline readings={kci.kci_readings ?? []} status={status} /></td>
                  </tr>
                )
              })}
            </tbody>
          </table>
        </div>
      )}
    </div>
  )
}
```

```typescript
import { redirect } from 'next/navigation'
import { createClient } from '@/lib/supabase/server'
import { KciForm } from './KciForm'
import type { AppRole } from '@/types/auth'

export const dynamic = 'force-dynamic'
export const metadata = { title: 'Create Key Control Indicator — GRC-Nexus' }

const CREATE_ROLES: AppRole[] = ['admin', 'audit-officer']

export default async function NewKciPage() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) redirect('/login')

  const appMeta    = user.app_metadata as Record<string, string>
  const activeRole = appMeta?.active_role as AppRole | undefined
  if (!activeRole || !CREATE_ROLES.includes(activeRole)) redirect('/audit/kcis')

  const [{ data: treatments }, { data: owners }] = await Promise.all([
    supabase.from('risk_treatments').select('id, title').order('title'),
    supabase.from('user_profiles').select('id, first_name, last_name').order('last_name'),
  ])

  const treatmentOptions = (treatments ?? []).map((t) => ({ value: t.id, label: t.title }))
  const ownerOptions = (owners ?? []).map((u) => ({
    value: u.id,
    label: `${u.first_name} ${u.last_name}`,
  }))

  return (
    <div className="max-w-2xl">
      <div className="mb-6">
        <h1 className="text-[20px] font-semibold text-navy-900 font-body">Create Key Control Indicator</h1>
        <p className="mt-1 text-[14px] text-navy-mid">Define a measurable metric to track control effectiveness.</p>
      </div>
      <KciForm treatments={treatmentOptions} owners={ownerOptions} />
    </div>
  )
}
```

```typescript
import { redirect } from 'next/navigation'
```
